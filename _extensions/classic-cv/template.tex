% create a coloured box with arrow and title as cv section headline
% param 1: section title
%
\newcommand{\cvsection}[1]
{
	\begin{center}
		\large\textcolor{sectcol}{\textbf{#1}}
	\end{center}
	\vspace{\cvsectionspacing}
}

%
\newcommand{\metasection}[2]
{
\footnotesize{#2} \hspace*{\fill} \footnotesize{#1}\\[1pt]
}

% New command for metasection without title
\newcommand{\metasectionnotitle}[2]
{
\footnotesize{#1} \hspace*{\fill} \footnotesize{#2}\\[1pt]
}

%----------------------------------------------------------------------------------------
%	 CV EVENT
%----------------------------------------------------------------------------------------

% creates a stretched box as cv entry headline followed by some paragraphs about
% the work you did
% param 1:	event time i.e. 2014 or 2011-2014 etc.
% param 2:	event name (what did you do?)
% param 3:	institution (where did you work / study)
% param 4:	list of paragraphs
%
\newcommand{\cvevent}[4]
{
\noindent

\begin{tabular*}{1\textwidth}{p{13.6cm}  x{3.9cm}}
	\textbf{#2} - \textcolor{bgcol}{#3} &   \vspace{2.5pt}\textcolor{sectcol}{#1}
\end{tabular*}

\vspace{-8pt}
\hspace*{\fill}\hspace*{-\hrulewidth}\textcolor{softcol}{\rule{\hrulewidth}{0.4pt}}\\
{\parskip=0pt
\vspace{-10pt}
\foreach \desc in {#4}{
	\par\noindent *  \desc\\[-10pt]
}}

\vspace{3pt}
}

% creates a stretched box as cv entry headline with logo followed by some paragraphs about
% the work you did
% param 1:	event time i.e. 2014 or 2011-2014 etc.
% param 2:	event name (what did you do?)
% param 3:	institution (where did you work / study)
% param 4:	list of paragraphs
% param 5:	logo path
%
\newcommand{\cveventlogo}[5]
{
\noindent

\begin{tabular*}{1\textwidth}{p{0.8cm} p{12.8cm}  x{3.9cm}}
	\ifthenelse{\equal{#5}{}}{}{
		\raisebox{-0.3\height}{\includegraphics[height=0.8cm]{#5}}
	} &
	\textbf{#2} - \textcolor{bgcol}{#3} &   \vspace{2.5pt}\textcolor{sectcol}{#1}
\end{tabular*}

\vspace{-8pt}
\hspace*{\fill}\hspace*{-\hrulewidth}\textcolor{softcol}{\rule{\hrulewidth}{0.4pt}}\\
{\parskip=0pt
\vspace{-10pt}
\foreach \desc in {#4}{
	\par\noindent *  \desc\\[-10pt]
}}

\vspace{3pt}
}
